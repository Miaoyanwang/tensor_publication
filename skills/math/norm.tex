\section*{Frobenius Norm}
\begin{lem}[Frobenius norm of product of matrices] For arbitrary two matrices, $A \in \mathbb{R}^{m \times r}$ and $B \in \mathbb{R}^{r \times n}$, we have
\[ \Fnorm{AB} \leq \norm{A} \Fnorm{B} , \]
	where $\norm{A}$ is the spectral norm or operator norm of $A$.
\end{lem}

\begin{proof}
	First, the spectral norm of $A \in \mathbb{R}^{m \times r}$ is defined as:
	\[ \norm{A} = \max_{x \in \mathbb{R}^r, \norm{x} \leq 1} \norm{Ax}.  \]
	Therefore, we have $\norm{Ax} \leq \norm{A} \norm{x}$ for $\forall x \in \mathbb{R}^r $. Let $B = [b_1,\dots, b_n] \in \mathbb{R}^{r \times n}$, where $b_j \in \mathbb{R}^{r}, j \in [n]$ are the columns of $B$. Then we have
	\[ \Fnorm{AB} = \sum_{j = 1}^n \norm{Ab_j} \leq  \norm{A}\sum_{j = 1}^n \norm{b_j} = \norm{A} \Fnorm{B} . \]
\end{proof}
